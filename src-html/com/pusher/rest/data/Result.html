001    package com.pusher.rest.data;
002    
003    import static com.pusher.rest.data.Result.Status.*;
004    
005    import java.io.IOException;
006    
007    import org.apache.http.client.ClientProtocolException;
008    
009    public class Result {
010        public enum Status {
011            SUCCESS(false), // No need!
012            CLIENT_ERROR(false),
013            AUTHENTICATION_ERROR(false),
014            MESSAGE_QUOTA_EXCEEDED(false),
015            NOT_FOUND(false),
016            SERVER_ERROR(true),
017            NETWORK_ERROR(true),
018            UNKNOWN_ERROR(true),
019            ;
020    
021            private final boolean shouldRetry;
022    
023            private Status(final boolean shouldRetry) {
024                this.shouldRetry = shouldRetry;
025            }
026    
027            /**
028             * Whether the call should be retried without modification with an expectation of success
029             */
030            public boolean shouldRetry() {
031                return shouldRetry;
032            }
033        }
034    
035        private final Status status;
036        private final Integer httpStatus;
037        private final String message;
038    
039        private Result(final Status status, final Integer httpStatus, final String message) {
040            this.status = status;
041            this.httpStatus = httpStatus;
042            this.message = message;
043        }
044    
045        public static Result fromHttpCode(final int statusCode, final String responseBody) {
046            final Status status;
047            switch (statusCode) {
048            case 200:
049                status = SUCCESS;
050                break;
051            case 400:
052                status = CLIENT_ERROR;
053                break;
054            case 401:
055                status = AUTHENTICATION_ERROR;
056                break;
057            case 403:
058                status = MESSAGE_QUOTA_EXCEEDED;
059                break;
060            case 404:
061                status = NOT_FOUND;
062                break;
063            default:
064                status = statusCode >= 500 && statusCode < 600 ? SERVER_ERROR : UNKNOWN_ERROR;
065            }
066    
067            return new Result(status, statusCode, responseBody);
068        }
069    
070        public static Result fromException(final IOException e) {
071            return new Result(Status.NETWORK_ERROR, null, e.toString());
072        }
073    
074        public static Result fromException(final ClientProtocolException e) {
075            return new Result(Status.UNKNOWN_ERROR, null, e.toString());
076        }
077    
078        /**
079         * Get the enum classifying the result of the call
080         */
081        public Status getStatus() {
082            return status;
083        }
084    
085        /**
086         * Get the data response (success) or descriptive message (error) returned from the call
087         */
088        public String getMessage() {
089            return message;
090        }
091    
092        /**
093         * Get the HTTP status code of the call, useful for debugging instances of UNKNOWN_ERROR
094         */
095        public Integer getHttpStatus() {
096            return httpStatus;
097        }
098    }
